Text animation


    let header = document.querySelectorAll('.header-name')

    window.addEventListener('load', () => {
        for (let i = 0; i < header.length; i++) {

            setTimeout( () => { header[i].style.opacity = '1'}, 100 * i);
            setTimeout( () => { header[i].style.transition = 'all 1s ease-out'}, 100 * i);
            
                if (i === header.length - 1) {

                    for (let i = 0; i < header.length; i++) {
                        
                        //removes the transition property after it's finished
                        setTimeout( () => { header[i].style.transition = ''}, 100 * i); 
                    };
                };
        };
    });

The javascript code for the fade in animation on the text on the index page.

A soundboard


    const sound1 = document.querySelector('.sound-1')
    const sound2 = document.querySelector('.sound-2')


    document.addEventListener('keydown', logKey);

    function logKey(e) {

        if (e.keyCode === 49) {

            //digit 1

            sound1.play();

        } if (e.keyCode === 50) {

            //digit 2

            sound2.play();

        }
    };

A snippet of code from a soundboard I made for myself. Press the right key, and a sound effect that's linked to it will play.
